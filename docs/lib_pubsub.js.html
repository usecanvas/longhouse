'use strict';

const createRedisClient = require('./create-redis-client');
const logfmt            = require('logfmt');
const sendMessage       = require('./send-message');
const spaces            = {};
const subscription      = createRedisClient();

/**
 * A module for subscribing to and publishing Longhouse space events.
 * @module pubsub
 */

configureSubscription(subscription);

module.exports = {
  flushSpaces : flushSpaces,
  joinClient  : joinClient,
  removeClient: removeClient,
};

/**
 * Empty out the spaces->clients object.
 *
 * This is for use in testing only, when it must be emptied between tests.
 */
function flushSpaces() {
  for (let key in spaces) {
    delete spaces[key];
  }
}

/**
 * Configure the pubsub subscription channel.
 *
 * The subscription channel listens for events published when client messages
 * are received.
 *
 * @private
 * @param {redis.Client} subscription The subscription channel to configure
 */
function configureSubscription(subscription) {
  subscription.psubscribe('spaces.*');

  subscription.on('pmessage', (_pattern, channel, identity) => {
    const space     = channel.split('.').slice(1, 2).join('');
    const eventName = channel.split('.').slice(2).join('.');

    switch (eventName) {
      case 'join':
        publishEvent('remote join', space, identity);
        break;
      case 'expire':
        publishEvent('remote expire', space, identity);
        break;
      case 'leave':
        publishEvent('remote leave', space, identity);
        break;
      default:
        logfmt.log({ event: `Unrecognized event: "${eventName}".` });
    }
  });
}

/**
 * Join a WebSocket client with an identity on a space.
 *
 * @param {String} space The space to join the client to
 * @param {ws.WebSocket} client The client joining the space
 */
function joinClient(space, client) {
  spaces[space] = spaces[space] || [];
  spaces[space].push(client);
  client.on('close', () => removeClient(client));
}

/**
 * Remove a client from all spaces or a given space.
 *
 * @param {ws.WebSocket} client The client leaving the space
 * @param {String} [space] The space to remove the client from
 */
function removeClient(client, space) {
  const timer = logfmt.time();

  let spaceCount = 0;

  if (space) {
    spaceCount++;
    removeClientFromSpace(client, space);
  } else {
    for (space in spaces) {
      spaceCount++;
      removeClientFromSpace(client, space);
    }
  }

  timer.log({ event: 'removed client', space_count: spaceCount });
}

/**
 * Remove a client from a given space.
 *
 * @private
 * @param {ws.WebSocket} client The client leaving the space
 * @param {String} space The space to remove the client from
 */
function removeClientFromSpace(client, space) {
  const clients = spaces[space];
  const index   = clients.indexOf(client);

  if (index !== -1) {
    clients.splice(index, 1);
  }
}

/**
 * Publish the given event name and associated identity to the space.
 *
 * This will not publish to any clients whose `__identity` matches the given
 * identity.
 *
 * @private
 * @param {String} eventName The name of the event to publish
 * @param {String} space The name of the space where the event occurred
 * @param {String} identity The identity associated with the event
 */
function publishEvent(eventName, space, identity) {
  const clients     = spaces[space] || [];
  const timer       = logfmt.time();

  let clientCount = 0;

  for (let i = 0, len = clients.length; i < len; i++) {
    const client = clients[i];

    if (client.__meta.identity === identity) {
      continue;
    }

    clientCount++;

    sendMessage(client,
      { action: eventName, space: space, identity: identity });
  }

  timer.log({
    event       : 'event published',
    event_name  : eventName,
    client_count: clientCount
  });
}
